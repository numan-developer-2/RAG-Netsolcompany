# Patterns to skip — noise chunks
SKIP_URLS = ['/tag/', '/category/', '/page/', '/feed/',
'/wp-admin/', '#', 'javascript:', 'mailto:']
def should_skip(chunk: dict) -> bool:
"""Return True if chunk is noise — skip from ChromaDB."""
# Too short
if chunk.get('word_count', 0) < MIN_CHUNK_WORDS:
return True
# Empty text
if not chunk.get('text', '').strip() or len(chunk.get('text','')) < 100:
return True
# Noise URLs (nav, pagination, feeds)
url = chunk.get('source_url', '')
if any(p in url for p in SKIP_URLS):
return True
# No title = probably a utility page
if not chunk.get('page_title', '').strip():
return True
return False
def recency_score(date_str: str) -> float:
"""0.5 (old) to 1.0 (recent). Used to boost fresh content."""
try:
d = datetime.fromisoformat(date_str.replace('Z',''))
days_old = (datetime.utcnow() - d).days
return round(max(0.5, 1.0 - (days_old / 365) * 0.5), 3)
except:
return 0.7
def build_metadata(chunk: dict, source_type: str) -> dict:
"""Build ChromaDB metadata — only str/int/float/bool allowed."""
meta = {
'chunk_id': str(chunk.get('chunk_id', '')),
'source_type': source_type,
'source_url': str(chunk.get('source_url', ''))[:300],
'page_title': str(chunk.get('page_title', ''))[:200],
'parent_url': str(chunk.get('parent_url', ''))[:300],
'level': int(chunk.get('level', 0)),
'scraped_date': str(chunk.get('scraped_date', '')),
'recency_score': recency_score(chunk.get('scraped_date','')),
'word_count': int(chunk.get('word_count', 0)),
}
# Extra fields for PDFs
if source_type == 'pdf':
meta['pdf_name'] = str(chunk.get('pdf_name', ''))[:200]
meta['page_number'] = int(chunk.get('page_number', 0))
meta['section_heading'] = str(chunk.get('section_heading',''))[:200]
meta['doc_category'] = str(chunk.get('doc_category', ''))
# Replace None with empty string
return {k: (v if v is not None else '') for k, v in meta.items()}

def ingest_web_pages():
col = chroma.get_or_create_collection(
name=COLLECTION_WEB,
metadata={'hnsw:space': 'cosine'} # cosine similarity
)
print(f'Existing: {col.count()} web chunks')
# Load + filter
chunks = []
with jsonlines.open(CHUNK_JSONL) as r:
for chunk in r:
if chunk.get('source_type') == 'web_page':
if not should_skip(chunk):
chunks.append(chunk)
print(f'After filter: {len(chunks)} web chunks')
# Batch embed + upsert
for i in tqdm(range(0, len(chunks), BATCH_SIZE), desc='Web Pages'):
batch = chunks[i : i + BATCH_SIZE]
texts = [c['text'] for c in batch]
try:
embeddings = embed_documents(texts)
col.upsert(
ids=[c['chunk_id'] for c in batch],
embeddings=embeddings,
documents=texts,
metadatas=[build_metadata(c, 'web_page') for c in batch]
)
time.sleep(0.05) # polite delay
except Exception as e:
print(f'Batch {i} failed: {e} — retrying one-by-one...')
time.sleep(10)
for chunk in batch:
try:
emb = embed_documents([chunk['text']])
col.upsert(
ids=[chunk['chunk_id']],
embeddings=emb,
documents=[chunk['text']],
metadatas=[build_metadata(chunk, 'web_page')]
)
time.sleep(0.5)
except Exception as e2:
print(f' Skip {chunk["chunk_id"]}: {e2}')
print(f'Web DONE. Total: {col.count()}')

def hybrid_retriever(state: RAGState) -> RAGState:
route = state.get('route', 'hybrid')
pdf_pri = state.get('pdf_priority', False)
all_results = []
seen_ids = set()
# ■■ Vector Search (all 3 query variants) ■■■■■■■■■■■■■■■■■
for query_variant in state['rewritten_queries']:
q_emb = embed_query(query_variant)
# Pick collections based on route
collections = []
if route in ['web_only', 'hybrid', 'competitive_mode']:
collections.append(('web', col_web))
if route in ['pdf_only', 'hybrid'] or pdf_pri:
collections.append(('pdf', col_pdf))
for ctype, col in collections:
# Build ChromaDB where filter
where = {}
mf = state.get('metadata_filters', {})
if mf.get('min_date'):
where['scraped_date'] = {'$gte': mf['min_date']}
res = col.query(
query_embeddings=[q_emb],
n_results=10,
where=where if where else None,
include=['documents', 'metadatas', 'distances']
)
for i, doc in enumerate(res['documents'][0]):
meta = res['metadatas'][0][i]
cid = meta.get('chunk_id', f'{ctype}_{i}')
score = 1.0 - res['distances'][0][i] # cosine: 1=identical
# Recency boost
score *= (0.7 + 0.3 * float(meta.get('recency_score', 0.7)))
# PDF authority boost
if pdf_pri and ctype == 'pdf':
score *= 1.4
if cid not in seen_ids:
seen_ids.add(cid)
all_results.append({
'chunk_id': cid,
'text': doc,
'source_url': meta.get('source_url', ''),
'page_title': meta.get('page_title', ''),
'source_type': ctype,
'pdf_name': meta.get('pdf_name', ''),
'page_number': meta.get('page_number', 0),
'rrf_score': score,
})

def multi_hop_retriever(state: RAGState) -> RAGState:
# Only runs for complex multi-hop queries
if state.get('complexity') != 'multi_hop':
return state
top5 = state['retrieved_chunks'][:5]
context = ' '.join([c['text'][:300] for c in top5])
existing= {c['chunk_id'] for c in state['retrieved_chunks']}
# Extract entities from first retrieval
sys = 'Extract key entities. Return ONLY a JSON array of 3 strings.'
usr = f'Query: {state["query"]}\nContext: {context}'
entities = llm_json(sys, usr)
if not isinstance(entities, list):
entities = list(entities.values())[:3]
extra = []
for entity in entities:
emb = embed_query(str(entity))
for col in [col_web, col_pdf]:
res = col.query(query_embeddings=[emb], n_results=4,
include=['documents','metadatas','distances'])
for i, doc in enumerate(res['documents'][0]):
meta = res['metadatas'][0][i]
cid = meta.get('chunk_id', '')
if cid and cid not in existing:
existing.add(cid)
extra.append({
'chunk_id': cid,
'text': doc,
'source_url': meta.get('source_url',''),
'page_title': meta.get('page_title',''),
'source_type': meta.get('source_type',''),
'pdf_name': meta.get('pdf_name',''),
'page_number': meta.get('page_number', 0),
'rrf_score': 1 - res['distances'][0][i],
})
combined = state['retrieved_chunks'] + extra
combined.sort(key=lambda x: x['rrf_score'], reverse=True)
return {**state, 'retrieved_chunks': combined[:TOP_K_RETRIEVE]}

from langgraph.graph import StateGraph, END
from state import RAGState
from nodes import (
query_analyzer, hybrid_retriever, multi_hop_retriever,
reranker_node, generator_node, hallucination_guard,
response_formatter
)
# ■■ Conditional routing functions ■■■■■■■■■■■■■■■■■■■■■■■■■■■■
def route_complexity(state: RAGState) -> str:
return 'multi_hop' if state.get('complexity')=='multi_hop' else 'rerank'
def route_guard(state: RAGState) -> str:
action = state.get('hallucination_action','USE_AS_IS')
if action == 'REGENERATE' and state.get('retry_count',0) < 2:
return 'retry'
return 'format'
# ■■ Build graph ■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■
g = StateGraph(RAGState)
g.add_node('query_analyzer', query_analyzer)
g.add_node('hybrid_retriever', hybrid_retriever)
g.add_node('multi_hop_retriever', multi_hop_retriever)
g.add_node('reranker', reranker_node)
g.add_node('generator', generator_node)
g.add_node('hallucination_guard', hallucination_guard)
g.add_node('response_formatter', response_formatter)
g.set_entry_point('query_analyzer')
g.add_edge('query_analyzer', 'hybrid_retriever')
g.add_conditional_edges('hybrid_retriever', route_complexity,
{'multi_hop':'multi_hop_retriever','rerank':'reranker'})
g.add_edge('multi_hop_retriever', 'reranker')
g.add_edge('reranker', 'generator')
g.add_edge('generator', 'hallucination_guard')
g.add_conditional_edges('hallucination_guard', route_guard,
{'retry':'hybrid_retriever','format':'response_formatter'})
g.add_edge('response_formatter', END)
rag_graph = g.compile()

from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from typing import List, Optional
import json, asyncio
from graph import run_query
from nodes import col_web, col_pdf
from config import LLM_MODEL
app = FastAPI(title='Netsol RAG API', version='2.0')
app.add_middleware(CORSMiddleware, allow_origins=['*'],
allow_methods=['*'], allow_headers=['*'])
class QueryRequest(BaseModel):
query: str
persona: Optional[str] = 'general'
chat_history: Optional[List[dict]]= []
# POST /query — standard response
@app.post('/query')
async def query_endpoint(req: QueryRequest):
try:
return run_query(req.query, req.persona, req.chat_history)
except Exception as e:
raise HTTPException(status_code=500, detail=str(e))
# POST /query/stream — SSE token streaming for frontend
@app.post('/query/stream')
async def stream_endpoint(req: QueryRequest):
async def generate():
yield f'data: {json.dumps({"stage":"analyzing"})}\n\n'
await asyncio.sleep(0.1)
result = run_query(req.query, req.persona, req.chat_history)
yield f'data: {json.dumps({"stage":"streaming"})}\n\n'
for word in result.get('answer','').split(' '):
yield f'data: {json.dumps({"stage":"token","token":word+" "})}\n\n'
await asyncio.sleep(0.02)
yield f'data: {json.dumps({"stage":"complete","result":result})}\n\n'
return StreamingResponse(generate(), media_type='text/event-stream')
# GET /health
@app.get('/health')
async def health():
return {'status':'ok','web_chunks':col_web.count(),
'pdf_chunks':col_pdf.count(),'model':LLM_MODEL}
